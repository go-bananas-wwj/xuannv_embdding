- [ ] **Step 1: 编写 prepare_aef_embeddings.py**

```python
import json
import argparse
from pathlib import Path
import numpy as np
import xarray as xr
import rasterio
from rasterio.transform import from_bounds
from rasterio.warp import reproject, Resampling
from affine import Affine
import torch

AEF_ZARR_URL = "s3://us-west-2.opendata.source.coop/tge-labs/aef-mosaic/"
AEF_TIME_MAP = {2017: 0, 2018: 1, 2019: 2, 2020: 3, 2021: 4, 2022: 5, 2023: 6, 2024: 7, 2025: 8}


def dequantize(aef_int8):
    """AEF int8 -> float32 dequantization."""
    return ((aef_int8.astype(np.float32) / 127.5) ** 2) * np.sign(aef_int8.astype(np.float32))


def read_aef_for_bbox(ds, bbox, year=2025):
    """
    bbox: (min_lon, min_lat, max_lon, max_lat)
    Returns (64, H, W) float32 array and its geotransform (WGS84).
    """
    time_idx = AEF_TIME_MAP[year]
    data = ds["embeddings"].isel(time=time_idx).sel(
        x=slice(bbox[0], bbox[2]),
        y=slice(bbox[3], bbox[1])  # y is likely descending
    ).compute()
    emb = data.values  # (band, y, x)
    emb = dequantize(emb)
    lon = data.x.values
    lat = data.y.values
    transform = Affine.translation(lon[0], lat[0]) * Affine.scale(lon[1] - lon[0], lat[1] - lat[0])
    return emb, transform


def resample_to_patch(aef_emb, aef_transform, patch_bounds, patch_shape=(128, 128)):
    """
    aef_emb: (64, H, W)
    patch_bounds: (minx, miny, maxx, maxy) in EPSG:4326
    Returns (64, 128, 128) float32 resampled to patch grid.
    """
    minx, miny, maxx, maxy = patch_bounds
    dst_transform = from_bounds(minx, miny, maxx, maxy, patch_shape[1], patch_shape[0])
    dst = np.empty((aef_emb.shape[0], *patch_shape), dtype=np.float32)
    reproject(
        source=aef_emb,
        destination=dst,
        src_transform=aef_transform,
        src_crs="EPSG:4326",
        dst_transform=dst_transform,
        dst_crs="EPSG:4326",
        resampling=Resampling.bilinear,
        num_threads=4,
    )
    return dst


def process_region(region, patch_meta_json, out_root):
    ds = xr.open_zarr(AEF_ZARR_URL, storage_options={"anon": True})
    out_root = Path(out_root) / region
    out_root.mkdir(parents=True, exist_ok=True)

    with open(patch_meta_json) as f:
        patches = json.load(f)

    for patch_id, meta in patches.items():
        bbox = meta["bbox"]  # [minx, miny, maxx, maxy]
        try:
            emb, transform = read_aef_for_bbox(ds, bbox, year=2025)
            patch_emb = resample_to_patch(emb, transform, bbox, patch_shape=(128, 128))
            # Save as (64, 128, 128) torch tensor, matching existing .pt layout
            torch.save(torch.from_numpy(patch_emb), out_root / f"{patch_id}_202512_embedding_map.pt")
            print(f"Saved {region}/{patch_id}_202512_embedding_map.pt")
        except Exception as e:
            print(f"Failed {patch_id}: {e}")


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--region", choices=["harbin", "haidian"], required=True)
    parser.add_argument("--patch-meta", required=True, help="JSON with patch_id -> bbox")
    parser.add_argument("--out-root", default="/data/xuannv_embedding/embeddings/aef_official_2025_annual")
    args = parser.parse_args()
    process_region(args.region, args.patch_meta, args.out_root)
```

- [ ] **Step 2: 生成 patch metadata JSON**

Create: `/root/workspace/xuannv/downstreams/scripts/build_patch_meta.py`

```python
import json
from pathlib import Path
import rasterio

regions = {
    "harbin": "/data/xuannv_embedding/processed/harbin/masks",
    "haidian": "/data/xuannv_embedding/processed/haidian/masks",
}

for region, mask_dir in regions.items():
    meta = {}
    for mask_path in Path(mask_dir).glob("*.tif"):
        with rasterio.open(mask_path) as src:
            bounds = src.bounds
            meta[mask_path.stem] = {
                "bbox": [bounds.left, bounds.bottom, bounds.right, bounds.top],
                "crs": str(src.crs),
                "shape": src.shape,
            }
    out_path = f"/data/xuannv_embedding/embeddings/aef_official_2025_annual/{region}_patch_meta.json"
    with open(out_path, "w") as f:
        json.dump(meta, f, indent=2)
    print(f"Wrote {out_path} ({len(meta)} patches)")
```

### Task 5: 汇总 AEF Benchmark 结果并与 V1.0 对比

**Files:**
- Create: `/root/workspace/xuannv/downstreams/scripts/summarize_aef_benchmark.py`
- Create: `/data/xuannv_embedding/experiments/aef_benchmark/AEF_BENCHMARK_REPORT.md`

- [ ] **Step 1: 编写汇总脚本**

```python
import json
from pathlib import Path
import numpy as np

v1_path = "/data/xuannv_embedding/experiments/v1.0/all_tasks_summary_final.json"
aef_root = Path("/data/xuannv_embedding/experiments/aef_benchmark")

v1 = json.load(open(v1_path))

task_map = {
    "construction": "construction_aef_2025",
    "building_change": "building_change_aef_2025",
    "farm_change": "farm_change_aef_2025",
    "rubbish": "rubbish_aef_2025",
    "construction_joint": "construction_joint_aef_2025",
}

report_lines = ["# AEF 2025 vs V1.0 Embedding Benchmark\n", "| Task | V1.0 AUC | AEF AUC | V1.0 F1 | AEF F1 | V1.0 mIoU | AEF mIoU |", "|---|---|---|---|---|---|---|"]

for task, exp_name in task_map.items():
    summary_file = aef_root / f"{exp_name}_5fold" / "summary_5fold.json"
    if not summary_file.exists():
        print(f"Missing {summary_file}")
        continue
    aef = json.load(open(summary_file))
    v1_auc = v1[task]["auc_roc_mean"]
    v1_f1 = v1[task]["f1_best_mean"]
    v1_miou = v1[task].get("miou_mean", 0)
    aef_auc = aef["auc_roc_mean"]
    aef_f1 = aef["f1_best_mean"]
    aef_miou = aef.get("miou_mean", 0)
    report_lines.append(f"| {task} | {v1_auc:.3f} | {aef_auc:.3f} | {v1_f1:.3f} | {aef_f1:.3f} | {v1_miou:.3f} | {aef_miou:.3f} |")

report_path = aef_root / "AEF_BENCHMARK_REPORT.md"
with open(report_path, "w") as f:
    f.write("\n".join(report_lines))
print(f"Report written to {report_path}")
```

```python
import json
import matplotlib.pyplot as plt
import numpy as np
from pathlib import Path

v1 = json.load(open("/data/xuannv_embedding/experiments/v1.0/all_tasks_summary_final.json"))
aef_root = Path("/data/xuannv_embedding/experiments/aef_benchmark")
task_map = {
    "construction": "construction_aef_2025",
    "building_change": "building_change_aef_2025",
    "farm_change": "farm_change_aef_2025",
    "rubbish": "rubbish_aef_2025",
    "construction_joint": "construction_joint_aef_2025",
}

tasks = list(task_map.keys())
v1_auc, aef_auc = [], []
v1_f1, aef_f1 = [], []
for task, exp_name in task_map.items():
    aef = json.load(open(aef_root / f"{exp_name}_5fold" / "summary_5fold.json"))
    v1_auc.append(v1[task]["auc_roc_mean"])
    aef_auc.append(aef["auc_roc_mean"])
    v1_f1.append(v1[task]["f1_best_mean"])
    aef_f1.append(aef["f1_best_mean"])

x = np.arange(len(tasks))
width = 0.35
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(14, 5))
ax1.bar(x - width/2, v1_auc, width, label="V1.0")
ax1.bar(x + width/2, aef_auc, width, label="AEF 2025")
ax1.set_ylabel("AUC-ROC")
ax1.set_xticks(x)
ax1.set_xticklabels(tasks, rotation=15, ha="right")
ax1.legend()
ax1.set_title("AUC-ROC Comparison")

ax2.bar(x - width/2, v1_f1, width, label="V1.0")
ax2.bar(x + width/2, aef_f1, width, label="AEF 2025")
ax2.set_ylabel("F1_best")
ax2.set_xticks(x)
ax2.set_xticklabels(tasks, rotation=15, ha="right")
ax2.legend()
ax2.set_title("F1_best Comparison")

fig.tight_layout()
out_path = aef_root / "visualizations" / "aef_vs_v1.0_comparison.png"
out_path.parent.mkdir(parents=True, exist_ok=True)
fig.savefig(out_path, dpi=300)
print(f"Saved comparison plot to {out_path}")
```
