# V1.1 AEF 蒸馏快速验证实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在 744 个哈尔滨+海淀 patch 上，用 AEF 2025 年度 embedding 作为教师、以较小蒸馏权重（0.1）从 V1.0 stage2 encoder 继续微调 10 个 epoch，快速验证下游任务是否有提升。

**Architecture:** 在 `highres_fusion` 之前的 base embedding map 上对学生做 L2 归一化，与同样 L2 归一化后的 AEF 2025 年度教师 embedding 计算 MSE 蒸馏损失；仅监督 2025 年 1-12 月（index 0-11），保留原有重建与均匀性损失。训练数据使用哈尔滨 424 + 海淀 320 共 744 个 patch 的合并 manifest。

**Tech Stack:** PyTorch + torch_npu，xarray/zarr，rasterio，WANDB。

```python
def build_region_meta_from_manifest(
    manifest_path: Path,
    region_root: Path | None = None,
) -> dict[str, dict[str, Any]]:
    """Build metadata for every patch in a region manifest.

    Uses the first existing georeferenced source file for each patch to obtain
    bbox, crs and shape. This works for both labeled and unlabeled patches.
    """
    manifest_path = Path(manifest_path)
    root = region_root if region_root is not None else manifest_path.parent
    fallback = load_fallback_map(manifest_path, root)

    meta: dict[str, dict[str, Any]] = {}
    with open(manifest_path, encoding="utf-8") as f:
        entries = json.load(f)

    for entry in entries:
        patch_id = entry.get("patch_id")
        if patch_id is None:
            continue
        source_path = fallback.get(patch_id)
        if source_path is None:
            logger.warning("No georeferenced source found for %s", patch_id)
            continue
        meta[patch_id] = extract_patch_meta(source_path)

    return meta
```

def _load_statistics(self) -> None:
    regions = set()
    for entry in self.manifest:
        regions.add(entry.get("region", self.region or ""))
    if "" in regions and self.region:
        regions.discard("")
        regions.add(self.region)

        for source in self.sources:

            if region:

if (
    teacher_embedding_map is not None
    and self.distill_weight > 0.0
    and output.base_embedding_map is not None
):
    student = output.base_embedding_map[:, :self.distill_months]  # (B, M, D, H, W)
    B, M, D, H, W = student.shape
    # 教师只有年度 embedding，复制到 2025 的 M 个月
    teacher = teacher_embedding_map[:, None, :, :, :].expand(-1, M, -1, -1, -1)
    student_norm = F.normalize(student.reshape(B * M, D, H, W), p=2, dim=1).reshape(B, M, D, H, W)
    teacher_norm = F.normalize(teacher.reshape(B * M, -1, H, W), p=2, dim=1).reshape(B, M, -1, H, W)
    # 若教师通道数与学生不同，先投影；本项目 AEF 教师=64，学生 embed_dim=64，无需投影
    distill = F.mse_loss(student_norm, teacher_norm)
    total_recon = total_recon + self.distill_weight * distill

- [ ] **Step 1: 编写合并 manifest 脚本**

```python
#!/usr/bin/env python3
from __future__ import annotations

import argparse
import json
from pathlib import Path


def main() -> None:
    p = argparse.ArgumentParser()
    p.add_argument("--harbin-manifest", type=Path, required=True)
    p.add_argument("--haidian-manifest", type=Path, required=True)
    p.add_argument("--output", type=Path, required=True)
    args = p.parse_args()

    combined: list[dict] = []
    for region, manifest_path in [("harbin", args.harbin_manifest),
                                  ("haidian", args.haidian_manifest)]:
        with open(manifest_path, encoding="utf-8") as f:
            entries = json.load(f)
        for entry in entries:
            entry = dict(entry)
            entry["region"] = region
            # paths are currently relative to region dir; prefix with region
            for key, value in entry.items():
                if key == "patch_id" or not isinstance(value, list):
                    continue
                entry[key] = [f"{region}/{v}" for v in value]
            combined.append(entry)

    args.output.parent.mkdir(parents=True, exist_ok=True)
    with open(args.output, "w", encoding="utf-8") as f:
        json.dump(combined, f, ensure_ascii=False, indent=2)
    print(f"Wrote {len(combined)} patches to {args.output}")


if __name__ == "__main__":
    main()
```
